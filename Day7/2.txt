# comparison operator :-

# => it is used to compare two values.
# => it always returns a boolean value ( True / False ).

# Types of comparison operators :-
# 1. Equal to operator (==) :-
# 2. Not Equal to operator (!=) :-
# 3. Greater than operator (>) :-
# 4. Less than operator (<) :-
# 5. Greater than or Equal to operator (>=) :-
# 6. Less than or Equal to operator (<=) :-


first_number = 10
second_number = 20

print(first_number == second_number)  # false 
print("output: ", first_number == second_number)  # false

print("output: ",first_number != second_number)  # true
print("output: ",first_number > second_number)  # false
print("output: ",first_number < second_number)  # true


a = 15
b = 15

print("output: ",a >= b)  # true
print("output: ",a <= b)  # true
